青岛口罩预约助手

这是一个用JavaScript写的脚本，用于青岛口罩预约系统，自动选择药房，并输入信息。
需要手动填写验证码并点击提交
将开头变量改为你的信息，药房名称替换为目标药房即可。
建议挂载到油猴脚本使用

// ==UserScript==
// @name         青岛口罩预约助手
// @namespace    http://tampermonkey.net/
// @version      1.0
// @author       AAAsuka
// @match        *://kzyynew.qingdao.gov.cn:81/*
// ==/UserScript==

var str = new Date().getFullYear()+"-"+(new Date().getMonth()+1)+"-"+new Date().getDate()+" 08:30:00";
var aimTime = new Date(Date.parse(str.replace(/-/g,"/")));
var countDown = aimTime.getTime() - new Date().getTime() + 500;

if(new Date() < aimTime){
    setTimeout(function(){location.reload();}, countDown);
}

var write = setInterval(function(){
    try{
        if(document.querySelector("i.layui-icon.layui-icon-ok")){
            document.querySelector("i.layui-icon.layui-icon-ok").click();
            document.querySelector("*[data-name='利群商厦'] > .layui-col-xs2.action.text-right").click();
        }
        else if(document.getElementsByName('name')){
            document.getElementsByName('name')[0].value = '你的名字'
            document.getElementsByName('mobile')[0].value = '你的电话';
            document.getElementsByName('code')[0].value = '你的身份证';

            clearInterval(write);
        }
        else{
            location.reload();
        }
    }
    catch(e){
    }
},500);
